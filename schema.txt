generator client {
  provider = "prisma-client"
  output   = "../src/generated/prisma"
}

datasource db {
  provider = "sqlite"
  url      = env("DATABASE_URL")
}

enum UserRole {
  employee
  manager
  admin
}

enum UserStatus {
  active
  inactive
}

enum PointageStatus {
  normal
  late
  incomplete
}

enum AbsenceType {
  conge
  maladie
  autre
}

enum AbsenceStatus {
  pending
  approved
  rejected
}

enum CorrectionStatus {
  pending
  approved
  rejected
}

enum ActivityType {
  auth
  pointage
  absence
  user
  validation
}

model User {
  id        String       @id @default(cuid())
  email     String?       @unique
  username  String       @unique
  password  String
  firstName String
  lastName  String
  role      UserRole    @default(employee)
  department String?
  position   String?
  avatar     String?
  status     UserStatus  @default(active)

  pointages          Pointage[]
  absences           Absence[]
  correctionRequests CorrectionRequest[]
  activityLogs       ActivityLog[]
  sessions          Session[]

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

model Pointage {
  id       String         @id @default(cuid())
  userId   String
  user     User           @relation(fields: [userId], references: [id], onDelete: Cascade)

  date     DateTime
  entryTime String?
  exitTime  String?
  duration  Int
  status    PointageStatus
  isActive  Boolean       @default(false)

  correctionRequests CorrectionRequest[]

  createdAt DateTime @default(now())
}

model Absence {
  id        String        @id @default(cuid())
  userId    String
  user      User          @relation(fields: [userId], references: [id], onDelete: Cascade)

  type      AbsenceType
  startDate DateTime
  endDate   DateTime
  reason    String
  status    AbsenceStatus
  comment   String?

  createdAt DateTime @default(now())
}

model CorrectionRequest {
  id         String           @id @default(cuid())
  userId     String
  user       User             @relation(fields: [userId], references: [id], onDelete: Cascade)

  pointageId String
  pointage   Pointage         @relation(fields: [pointageId], references: [id], onDelete: Cascade)

  requestDate    DateTime
  originalEntry  String?
  originalExit   String?
  newEntry       String
  newExit        String
  reason         String
  status         CorrectionStatus

  createdAt DateTime @default(now())
}

model ActivityLog {
  id        String       @id @default(cuid())
  userId    String
  user      User         @relation(fields: [userId], references: [id], onDelete: Cascade)

  action    String
  details   String
  timestamp DateTime      @default(now())
  type      ActivityType
}

model SystemSettings {
  id               Int      @id @default(1)
  workStartTime    String
  workEndTime      String
  breakDuration    Int
  overtimeThreshold Int
  holidays         Json

  updatedAt DateTime @updatedAt
}

model Session {
  id           String   @id @default(cuid())
  userId       String
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  sessionToken String   @unique
  ipAddress    String?
  userAgent    String?
  expiresAt    DateTime

  createdAt    DateTime @default(now())
}
